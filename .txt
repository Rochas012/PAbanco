# 1 - Liste o número do funcionário, nome completo e o nome do departamento em que trabalha.
Select e.emp_no, e.first_name, e.last_name, d.dept_name
from employees e 
join dept_emp de ON e.emp_no = de.emp_no 
join departments d ON de.dept_no = d.dept_no;

# 2 - Exiba o nome do gerente, o número do funcionário e o nome do departamento que ele gerencia.
Select e.emp_no, e.first_name, e.last_name, d.dept_name
from employees e 
join dept_manager m ON e.emp_no = m.emp_no
join departments d ON d.dept_no = m.dept_no;

# 3 - Liste todos os funcionários e mostre o nome do departamento, mesmo que o funcionário não esteja alocado em nenhum departamento.
select e.emp_no, concat(e.first_name, ' ' , e.last_name) as nome_completo, d.dept_name
from employees e 
left join dept_emp de ON de.emp_no = e.emp_no
left join departments d ON d.dept_no = de.emp_no;
